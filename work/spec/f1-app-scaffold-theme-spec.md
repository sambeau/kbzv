### 3.2 vite.config.ts

**Exact content:**

```typescript
import { defineConfig } from "vite";
import react from "@vitejs/plugin-react";
import tailwindcss from "@tailwindcss/vite";
import path from "path";

// https://v2.tauri.app/start/frontend/vite/
const host = process.env.TAURI_DEV_HOST;

export default defineConfig(async () => ({
  plugins: [react(), tailwindcss()],

  resolve: {
    alias: {
      "@": path.resolve(__dirname, "./src"),
    },
  },

  // Vite options tailored for Tauri development
  clearScreen: false,
  server: {
    port: 1420,
    strictPort: true,
    host: host || false,
    hmr: host
      ? {
          protocol: "ws",
          host,
          port: 1421,
        }
      : undefined,
    watch: {
      // Tell Vite to ignore watching `src-tauri`
      ignored: ["**/src-tauri/**"],
    },
  },
}));
```

**Key points:**
- The `@` alias resolves to `./src` — this is required by shadcn/ui's import convention (`@/components/ui/...`, `@/lib/utils`)
- Tailwind CSS v4 uses the `@tailwindcss/vite` plugin instead of PostCSS
- Port `1420` is the Tauri convention; `strictPort: true` prevents fallback to another port
- `clearScreen: false` because Tauri manages the terminal output
- `src-tauri/` is excluded from the file watcher

## 4. Styling

### 4.1 Tailwind CSS v4 Configuration

Tailwind CSS v4 uses CSS-based configuration instead of a separate `tailwind.config.ts` file. All configuration is handled via `@theme` directives in `globals.css`.

**If the Tauri scaffold or shadcn init generates a `tailwind.config.ts`**, keep it. The shadcn/ui CLI may still generate one for compatibility. Its content will be minimal:

```typescript
import type { Config } from "tailwindcss";

export default {
  darkMode: "class",
  content: ["./index.html", "./src/**/*.{ts,tsx,js,jsx}"],
} satisfies Config;
```

**However, with Tailwind CSS v4 and the latest shadcn/ui,** configuration is more likely handled entirely in CSS. The `@tailwindcss/vite` plugin auto-discovers content files. Follow whatever `npx shadcn@latest init` generates — do not manually create a config file that conflicts with the generated output.

```typescript
 * - 'workflows': Entity tree and detail panel (Feature 3)
 */
type ActiveView = "documents" | "workflows";

/**
 * UI state for the application shell.
 * Grows in later features with selectedEntityId, navigationHistory, etc.
 */
interface UIState {
  /** Absolute path to the open Kanbanzai project, or null if none open */
  projectPath: string | null;

  /** Currently active top-level view */
  activeView: ActiveView;

  /** Set the project path. Pass null to close the project. */
  setProjectPath: (path: string | null) => void;

  /** Switch the active top-level view */
  setActiveView: (view: ActiveView) => void;
}

const useUIStore = create<UIState>((set) => ({
  // --- State ---
  projectPath: null,
  activeView: "documents",

  // --- Actions ---
  setProjectPath: (path) => set({ projectPath: path }),
  setActiveView: (view) => set({ activeView: view }),
}));

export { useUIStore };
export type { ActiveView, UIState };
```

**Behavioural constraints:**

| Aspect | Specification |
|--------|--------------|
| Default `projectPath` | `null` — no project open on launch |
| Default `activeView` | `"documents"` — Documents view is the landing tab |
| Persistence | None — store resets on app restart |
| Subscribers | Any component can subscribe via `useUIStore()` or `useUIStore((s) => s.field)` |
| Immutability | Zustand `set` produces a new state object; no mutation |
| Store name | `useUIStore` (hook convention) |

**Type exports:**
- `ActiveView` — union type, exported for use in component props
- `UIState` — full interface, exported for testing and type references
- `useUIStore` — the Zustand hook, default export candidate but named export preferred

```typescript
import { FileText, FolderOpen, GitBranch } from "lucide-react";
import { EmptyState } from "@/components/common/EmptyState";
import { useUIStore } from "@/lib/store/ui-store";
import { open } from "@tauri-apps/plugin-dialog";
import { exists } from "@tauri-apps/plugin-fs";
import { message } from "@tauri-apps/plugin-dialog";

async function handleOpenProject(setProjectPath: (path: string | null) => void) {
  // 1. Open native folder picker
  const selected = await open({
    directory: true,
    multiple: false,
    title: "Open Kanbanzai Project",
  });

  // 2. User cancelled
  if (selected === null) {
    return;
  }

  // 3. Validate: .kbz/config.yaml must exist
  const configPath = `${selected}/.kbz/config.yaml`;
  const isValid = await exists(configPath);

  if (isValid) {
    // 4a. Valid project — store the path
    setProjectPath(selected);
  } else {
    // 4b. Invalid — show native error dialog
    await message(
      "The selected folder does not contain a .kbz/config.yaml file. Please select a folder that was initialised with Kanbanzai.",
      {
        title: "Not a Kanbanzai Project",
        kind: "error",
      }
    );
  }
}

function MainPanel() {
  const activeView = useUIStore((s) => s.activeView);
  const projectPath = useUIStore((s) => s.projectPath);
  const setProjectPath = useUIStore((s) => s.setProjectPath);

  // No project open — show open prompt
  if (!projectPath) {
    return (
      <main className="flex-1 overflow-auto">
        <EmptyState
          icon={FolderOpen}
          title="No project open"
          description="Select a Kanbanzai project folder to get started"
          action={{
            label: "Open Project",
            onClick: () => handleOpenProject(setProjectPath),
          }}
        />
      </main>
    );
  }

  // Project open — show view-specific placeholder
  return (
    <main className="flex-1 overflow-auto">
      {activeView === "documents" ? (
        <EmptyState
          icon={FileText}
          title="Documents"
          description="Document list coming in a future update"
        />
      ) : (
        <EmptyState
          icon={GitBranch}
          title="Workflows"
          description="Workflow tree coming in a future update"
        />
      )}
    </main>
  );
}

export { MainPanel };
```

**Exact Tailwind classes on `<main>`:**

| Class | Purpose |
|-------|---------|
| `flex-1` | Fill remaining vertical space after header |
| `overflow-auto` | Scroll if content exceeds viewport |

**`handleOpenProject` function specification:**

This is a standalone `async` function (not a React hook or component method) that encapsulates the project open flow. It accepts `setProjectPath` as a parameter to avoid closing over the store hook.

| Step | Action | API | Error handling |
|------|--------|-----|---------------|
| 1 | Open native folder picker | `open({ directory: true, multiple: false, title: "Open Kanbanzai Project" })` | If `open()` throws, the error propagates unhandled (Tauri's dialog API is reliable) |
| 2 | Check cancellation | Return value is `null` | Return early, no action |
| 3 | Validate project | `exists(\`${selected}/.kbz/config.yaml\`)` | If `exists()` throws (FS error), let it propagate |
| 4a | Valid project | `setProjectPath(selected)` | N/A |
| 4b | Invalid project | `message(...)` with `kind: "error"` | Native dialog; dismiss returns to previous state |

**Edge cases:**

|----------|-----------|
| User cancels folder picker | `open()` returns `null`; function returns; no state change |

| Path contains spaces or special characters | Handled by Tauri's native dialog — returns properly escaped paths |

| Aspect | Specification |
|--------|--------------|
| Action | Toggle OS appearance to dark mode (macOS: System Preferences → Appearance → Dark) |
| Expected | The app's colours change — dark background, light text |

| Pass condition | Both light and dark modes render correctly; no elements are invisible (e.g., white text on white background) |

**Note on dark mode implementation:** The `prefers-color-scheme` detection requires one of:
1. The shadcn init sets up automatic dark mode detection (check generated globals.css), or
2. A small script in `index.html` or `main.tsx` that applies the `.dark` class based on `window.matchMedia('(prefers-color-scheme: dark)')` and listens for changes

If the shadcn init does not handle this automatically, add the following to `main.tsx` before `ReactDOM.createRoot`:

```typescript
// Sync dark mode with system preference
const darkModeMediaQuery = window.matchMedia("(prefers-color-scheme: dark)");
function applyDarkMode(e: MediaQueryListEvent | MediaQueryList) {
  document.documentElement.classList.toggle("dark", e.matches);
}
applyDarkMode(darkModeMediaQuery);
darkModeMediaQuery.addEventListener("change", applyDarkMode);
```
